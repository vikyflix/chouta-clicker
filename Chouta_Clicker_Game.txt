    *stormActive
        
        name: Highstorm Active
        desc: The Highstorm is raging. Production is altered.
        //icon: https://uploads.coppermind.net/thumb/The_Highstorm_by_Sakina_Minnis.jpg/200px-The_Highstorm_by_Sakina_Minnis.jpg

    *stormInActive
        
        name: Highstorm InActive
        desc: The Highstorm is raging. Production is altered.
        //icon: https://uploads.coppermind.net/thumb/The_Highstorm_by_Sakina_Minnis.jpg/200px-The_Highstorm_by_Sakina_Minnis.jpg

    *stormCounterTest
        hidden
        no text
        no buy
        start with: 1
        name: TEST
        desc: TESTING HIGHSTORM EVENT
        icon: https://uploads.coppermind.net/thumb/The_Highstorm_by_Sakina_Minnis.jpg/200px-The_Highstorm_by_Sakina_Minnis.jpg
        on tick:
          yield 1 stormActive

          if (stormlightStored >= spheres*1) 
                show stormlightBar0 
          else 
                hide stormlightBar0
          end
          
          if (stormlightStored >= spheres*5) 
                hide stormlightBar0
                show stormlightBar5 
          else 
                hide stormlightBar5
          end

          if (stormlightStored >= spheres*25)
                hide stormlightBar5
                show stormlightBar25 
          else 
                hide stormlightBar25
          end
          if (stormlightStored >= spheres*50) 
                hide stormlightBar25
                show stormlightBar50
          else 
                hide stormlightBar50
          end
          if (stormlightStored >= spheres*75) 
                hide stormlightBar50
                show stormlightBar75
          else 
                hide stormlightBar75
          end
          if (stormlightStored >= spheres*100)
                hide stormlightBar75
                show stormlightBar100
          else 
                hide stormlightBar100
          end
          if (stormActive==120)
            lose stormActive stormActive
            yield 1 stormInActive
          else if (stormInActive==5)
            toast STORM RISES

            lose stormActive stormActive
            lose  5 stormInActive
          end
          
          if (stormInActive==0 and stormlightStored < spheres*100)
                yield (spheres * 100) * min(1, (1 + betterSpheres) * (0.5 + stormCage*0.25)) stormlightStored        
          else
          end
          if (stormInActive==0)
            show stormCounterTest
            show stormlight
            show Background1
          else
            hide stormCounterTest
            hide stormlight
            hide Background1
          end
          
        end
