    *stormCounterTest
        hidden
        no text
        no buy
        start with: 1
        name: TEST
        desc: TESTING HIGHSTORM EVENT
        icon: https://uploads.coppermind.net/thumb/The_Highstorm_by_Sakina_Minnis.jpg/200px-The_Highstorm_by_Sakina_Minnis.jpg
        on tick:
          yield 20 stormActive
          if (stormlightStored >= spheres*25) show stormlightBar25 else hide stormlightBar25
          if (stormlightStored >= spheres*50) show stormlightBar50 else hide stormlightBar50
          if (stormlightStored >= spheres*75) show stormlightBar75 else hide stormlightBar75
          if (stormlightStored >= spheres*100) show stormlightBar100 else hide stormlightBar100
          if (stormActive==120)
            lose stormActive stormActive
            yield 1 stormInActive
          else if (stormInActive==5)
            toast STORM RISES
            lose stormActive stormActive
            lose  5 stormInActive
          end
          if (stormInActive==0)
            show stormCounterTest
            show stormlight
            if (stormlightStored < spheres*100)
                yield 20 stormlightStored
            else if (stormlightStored == spheres*100)
                toast "Your spheres are fully infused!"
            end
          else
            hide stormCounterTest
            hide stormlight
          end
        end
